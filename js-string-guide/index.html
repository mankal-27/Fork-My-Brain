// Real string problems, real methods
const username = "  Mankal_27  ";

// Clean up user input
username.trim();            // "Mankal_27"

// Validate an email
"hi@fork.dev".includes("@");  // true

// Build a URL slug from a title
"Hello World!"
  .toLowerCase()        // "hello world!"
  .replace(/[^a-z0-9]+/g, "-") // "hello-world-"
  .replace(/-$/, "");      // "hello-world"

// Truncate a long tweet
"JavaScript is awesome".slice(0, 10) + "...";
// "JavaScript..."
