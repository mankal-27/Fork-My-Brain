// Real problems, real methods
const cart = [
  { name: 'Shoes',  price: 2499 },
  { name: 'Hat',    price: 399  },
];
// Filter under ₹500
cart.filter(i => i.price < 500); // [{Hat}]
// Apply 10% discount
cart.map(i => ({...i, price: i.price * 0.9}));
// Calculate total
cart.reduce((s, i) => s + i.price, 0); // 2898
